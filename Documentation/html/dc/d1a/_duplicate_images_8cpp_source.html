427 FeatureVectorPtr DuplicateImage::ExampleWithSmallestScanLine ()
428 {
429  kkint32 smallestScanLine = 99999999;

431 
432 
433  for (FeatureVectorList::iterator iIDX = duplicatedImages.begin (); iIDX != duplicatedImages.end (); iIDX++)
434  {
435  FeatureVectorPtr i = *iIDX;
436  // First lets derive scan line from example file name
437 
438  KKStr rootName = osGetRootName (i->ExampleFileName ());
439  rootName.Upper ();
440 
441  kkint32 scanLine = 9999999;
442 
443  if (rootName.SubStrPart (0, 4) == "FRAME")
444  {
445  // Scan line will be last seq number in name.
446  kkint32 x = rootName.LocateLastOccurrence ('_');
447  if (x > 0)
448  {
449  KKStr scanLineStr = rootName.SubStrPart (x + 1);
450  scanLine = atoi (scanLineStr.Str ());
451  }
452  }
453  else
454  {
455  // Scan should be 2nd to last seq number in name.
456  kkint32 x = rootName.LocateLastOccurrence ('_');
457  if (x > 0)
458  {
459  KKStr workStr = rootName.SubStrPart (0, x - 1);
460  kkint32 x = workStr.LocateLastOccurrence ('_');
461  KKStr scanLineStr = workStr.SubStrPart (x + 1);
462  scanLine = atoi (scanLineStr.Str ());
463  }
464  }
465 
466  if ((scanLine < smallestScanLine) ||
467  (imageWithSmallestScanLine == NULL)
468  )
469  {
470  smallestScanLine = scanLine;

472  }
473  }
