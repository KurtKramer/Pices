436  VolConstBool& _cancelFlag,

441  // We do not bother with the base class 'TrainModel' like we do with other models.
442  // 'ModelOldSVM' is a special case. All we are trying to do is create a pass through
443  // for 'svmModel'.
444 
445  delete svmModel;
446  svmModel = NULL;
447 
448  Model::TrainModel (_trainExamples, _alreadyNormalized, _takeOwnership, _cancelFlag, _log);
449  // The "Model::TrainModel" may have manipulated the '_trainExamples'. It will have also

454  fileDesc = trainExamples->FileDesc ();
455 
456  SVMparamPtr svmParam = Param ()->SvmParameters ();
457  if (!svmParam)

459  validModel = false;
460  KKStr errMsg = " ModelOldSVM::TrainModel ***ERROR*** (svmParam == NULL).";
461  _log.Level (-1) << endl << errMsg << endl << endl;
462  throw KKException (errMsg);
463  }
464 
465  {
466  delete classes;
467  classes = trainExamples->ExtractListOfClasses ();
468  classes->SortByName ();
469  numOfClasses = classes->QueueSize ();
470  delete assignments;
471  assignments = new ClassAssignments (*classes);
472  }
473 
474  try
475  {
476  TrainingTimeStart ();
477  svmModel = new SVMModel (*svmParam, *trainExamples, *assignments, fileDesc, _log);
478  TrainingTimeEnd ();

483  validModel = false;

485  svmModel = NULL;
486  }
487 
488  if (weOwnTrainExamples)

587  }
588  }
589  delete t;
590  t = s.GetNextToken (log);
591  }
592 
593  if (!param)
594  param = dynamic_cast<ModelParamOldSVMPtr> (Model::param);
