143 void FrameProcessorThread::ProcessFrame (LogicalFramePtr frame)
144 {
145  if (TerminateFlag ())
146  return;
147 
148  ExtractedImageListPtr extractedImages = frame->ProcessFrame ();
149  if (!extractedImages)
150  return;
151 
152  imagesFound += frame->ImagesInFrame ();
153 
154  while ((extractedImages->QueueSize () > 0) && (!TerminateFlag ()))
155  {
156  ExtractedImagePtr extractedImage = extractedImages->PopFromBack ();
157 
158  kkuint32 sipperTopRow = extractedImage->SipperFileScanRow ();
159  kkuint32 sipperTopCol = extractedImage->SipperFileScanCol ();
160 
161  KKStr imageFileName (64);
162  imageFileName << sipperRootName
163  << "_"
164  << StrFormatInt (sipperTopRow, "ZZ00000000")
165  << "_"
166  << StrFormatInt (sipperTopCol, "0000")
167  << ".bmp";
168 
169  FeatureVectorPtr featureVector = NULL;
170  ImageFeaturesPtr picesFeatureVector = NULL;
171 
172  RasterSipperPtr raster = extractedImage->Image ();
173  raster->FileName (imageFileName);
174 
175  float depth = 0.0f;
176  MLClassPtr predClass1 = unKnownMLClass;
177  MLClassPtr predClass2 = NULL;
178  kkint32 predClass1Votes = 0;
179  kkint32 predClass2Votes = 0;
180  double knownClassProb = 0.0;
181  double predClass1Prob = 0.0;
182  double predClass2Prob = 0.0;
183 
184  bool cf = false;
185 
186 
187  /**@todo When a proper implementation for SIPPER 4 is made I will need to get instrument data for it. */
188 
189  InstrumentDataPtr id = NULL;
190  if (this->parms.FileFormat () != sfSipper4Bit)
191  id = InstrumentDataFileManager::GetClosestInstrumentData (imageFileName, cf, log);
192 
193  if (id)
194  depth = id->Depth ();
195 
196  if (parms.ExtractFeatureData ())
197  {
198  raster->FileName (imageFileName);
199  featureVector = fvProducer->ComputeFeatureVector (*raster, unKnownMLClass, NULL, 1.0f, log);
200  featureVector->ExampleFileName (imageFileName);
201 
202  picesFeatureVector = NULL;
203  if (typeid (*featureVector) == typeid (ImageFeatures))
204  {
205  picesFeatureVector = dynamic_cast<ImageFeaturesPtr> (featureVector);
206  featureVector = NULL;
207  }
208  else
209  {
210  picesFeatureVector = new ImageFeatures (*featureVector);
211  delete featureVector;
212  featureVector = NULL;
213  }
214 
215  if (id)
216  {
217  picesFeatureVector->FlowRate1 (id->FlowRate1 ());
218  picesFeatureVector->FlowRate2 (id->FlowRate2 ());
219  picesFeatureVector->Latitude (id->Latitude ());
220  picesFeatureVector->Longitude (id->Longitude ());
221  picesFeatureVector->Depth (id->Depth ());
222  picesFeatureVector->Salinity (id->Salinity ());
223  picesFeatureVector->Oxygen (id->Oxygen ());
224  picesFeatureVector->Fluorescence (id->Fluorescence ());
225  }
226 
227  picesFeatureVector->SfCentroidCol (picesFeatureVector->CentroidCol () + float (sipperTopCol));
228  picesFeatureVector->SfCentroidRow (picesFeatureVector->CentroidRow () + double (sipperTopRow));
229 
230  // Duplicate Image Detection was moved over to the 'DataBaseUpdateThread".
231 
232  if (classifier)
233  {
234  int numOfWinners;
235  bool knownClassOneOfTheWinners = false;
236  double breakTie = 0.0f;
237 
238  ImageFeaturesPtr dupFV = new ImageFeatures (*picesFeatureVector);
239  classifier->ClassifyAImage (*dupFV,
240  predClass1, predClass2,
241  predClass1Votes, predClass2Votes,
242  knownClassProb,
243  predClass1Prob, predClass2Prob,
244  numOfWinners,
245  breakTie
246  );
247 
248  picesFeatureVector->MLClass (predClass1);
249  picesFeatureVector->Probability ((float)predClass1Prob);
250  picesFeatureVector->BreakTie ((float)breakTie);
251  picesFeatureVector->PredictedClass (predClass1);
252  delete dupFV;
253  dupFV = NULL;
254  }
255  }
256 
257  extractedImage->Depth (depth);
258  extractedImage->FeatureVector (picesFeatureVector);
259 
260  extractedImage->Prediction (predClass1, (float)predClass1Prob, predClass2, (float)predClass2Prob);
261 
262  ++imagesClassified;
263 
264  imagesAwaitingUpdate->AddExtractedImage (extractedImage);
265  }
266 
267  delete extractedImages;
268  extractedImages = NULL;
269 
270  ++framesProcessed;
271  scanLinesProcessed += frame->FrameHeight ();
272 } /* ProcessFrame */

319 LogicalFramePtr FrameProcessorThread::GetNextFrameToProcess ()
320 {
321  LogicalFramePtr logicalFrame = framePool->GetNextFrameToProcess ();
322  while ((logicalFrame == NULL) && (!CancelFlag ()))
323  {
324  osSleepMiliSecs (20);
325  logicalFrame = framePool->GetNextFrameToProcess ();
326  if ((logicalFrame == NULL) && ShutdownFlag ())
327  {
328  // There are no more frames needing processing and the Extraction Manager wants to stop when convenient.
329  break;
330  }
331  }
332 
333  if (CancelFlag () && (logicalFrame != NULL))
334  {
335  framePool->AddBackToFramesAwaitingProcessing (logicalFrame);
336  logicalFrame = NULL;
337  }
338 
339  return logicalFrame;
340 } /* GetNextFrameToProcess */

346 {
347  Status (ThreadStatus::Starting);
348 
349  Status (ThreadStatus::Running);
350 
351  LogicalFramePtr logicalFrame = GetNextFrameToProcess ();
352  while (!CancelFlag ())
353  {
354  if (logicalFrame)
