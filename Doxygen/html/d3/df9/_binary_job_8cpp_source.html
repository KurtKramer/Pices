947  float inAccuracy
948  )
949 {
950  const float maxProbDesired = 99.999f;
951  const float power = 2.0f;
952  float prob = inProb;
953  if (prob > maxProbDesired)
954  {
955  float zed = (prob - maxProbDesired) / (100.0f - maxProbDesired);
956  prob = (pow (power, zed) - 1.0f) + maxProbDesired;
957  }
958 
959  return fabs (inAccuracy - prob);
960 }
