192  int32 MemoryConsumedEstimated () const
193  {
194  int32 memoryConsumedEstimated = sizeof (svm_model)
195  + param.MemoryConsumedEstimated ()
196  + (int32)exampleNames.size () * 40;
197 
198  if (SV) memoryConsumedEstimated += sizeof (svm_node*) * l;
199  if (sv_coef) memoryConsumedEstimated += (nr_class - 1) * sizeof (double*) + l * (nr_class - 1) * sizeof (double);
200  if (rho) memoryConsumedEstimated += l * sizeof (double);
201  if (label) memoryConsumedEstimated += nr_class * sizeof (int32);
202  if (nSV) memoryConsumedEstimated += nr_class * sizeof (int32);
203  if (featureWeight) memoryConsumedEstimated += dim * sizeof (double);
204  if (kValueTable) memoryConsumedEstimated += sizeof (double) * l;
205  if ((xSpace != NULL) && weOwnXspace)
206  memoryConsumedEstimated += sizeof (svm_node) * l;
207 
208  return memoryConsumedEstimated;
