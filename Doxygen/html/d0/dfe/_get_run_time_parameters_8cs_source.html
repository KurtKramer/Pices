512  private void GradeAgainstCrossValidationRun ()
513  {
514  if (curSelModel == null)
515  curSelModel = new PicesTrainingConfigManaged (configFileName, runLog);
516 
517  if (!curSelModel.Valid ())
518  {
519  MessageBox.Show ("Configuration File[" + configFileName + "] is not valid");
520  return;
521  }
522 
523  bool changesMade = false;
524  PicesFeatureVectorList data = curSelModel.LoadFeatureDataFromTrainingLibraries (changesMade);
525  if (cancelProcessing)
526  {
527  data = null;
528  return;
529  }
530 
531  if (data == null)
532  {
533  MessageBox.Show ("No data was loaded from Configuration File[" + configFileName + "] is not valid");
534  return;
535  }
536 
537  PicesFeatureVectorList examples = data.StratifyAmoungstClasses (numOfFolds);
538 
539  int imageCount = examples.Count;
540  int numImagesPerFold = (imageCount + numOfFolds - 1) / numOfFolds;
541 
542  int firstInGroup = 0;
543 
544  int foldNum;
545 
546  for (foldNum = 0; foldNum < numOfFolds; foldNum++)
547  {
548  int lastInGroup;
549 
550  // If We are doing the last Fold Make sure that we are including all the examples
551  // that have not been tested.
552  if (foldNum == (numOfFolds - 1))
553  lastInGroup = imageCount;
554  else
555  lastInGroup = firstInGroup + numImagesPerFold - 1;
556 
557  PicesFeatureVectorList trainingImages = new PicesFeatureVectorList ();
558  PicesFeatureVectorList testImages = new PicesFeatureVectorList ();
559 
560  for (int x = 0; (x < imageCount) && (!cancelProcessing); x++)
561  {
562  PicesFeatureVector newImage = new PicesFeatureVector (examples[x]);
563 
564  if ((x >= firstInGroup) && (x <= lastInGroup))
565  testImages.Add (newImage);
566  else
567  trainingImages.Add (newImage);
568  }
569 
570  if (cancelProcessing)
571  break;
572 
573  CrossValidate (testImages, trainingImages, foldNum);
574 
575  firstInGroup = firstInGroup + numImagesPerFold;
576  }
577 
578  if (!cancelProcessing)
579  {
580  if (trainingModel != null)
581  {
582  trainingModel = null;
583  }
584 
585  msgQueue.AddMsg ("\n" + "Building Final Classifier for User Query");
586  trainingModel = new TrainingModel2 (runLog, curSelModel);
587  trainingModel.BuildTrainingModel (data);
588  }
589 
590  if ((!cancelProcessing) && (trainingModel != null) && (!trainingModel.Valid))
591  {
592  MessageBox.Show ("Error Loading Training Model; refer to dialog box");
593  cancelProcessing = true;
594  }
595 
596  else if (!cancelProcessing)
597  {
598  confusionMatrix.SetTrainingModel (trainingModel);
599  }
600 
601  // We no longer need the Training Model or Ground Truth loaded
602  BlockStart ();
603  data = null;
604  BlockEnd ();
605  } /* GradeAgainstCrossValidationRun */
