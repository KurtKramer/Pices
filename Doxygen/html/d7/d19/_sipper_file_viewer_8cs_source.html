838  private void ComputeOutlierBounds (float[] data,
839  ref float lowerBound,
840  ref float upperBound
841  )
842  {
843  if ((data == null) || (data.Length < 1))
844  return;
845 
846  int x = 0;
847 
848  float total = 0.0f;
849  for (x = 0; x < data.Length; ++x)
850  total += data[x];
851  float mean = total / data.Length;
852 
853  double totalDeltaSquared = 0.0f;
854  for (x = 0; x < data.Length; ++x)
855  {
856  float delta = data[x] - mean;
857  double deltaSquared = delta * delta;
858  totalDeltaSquared += deltaSquared;
859  }
860 
861  double variance = totalDeltaSquared / (double)data.Length;
862  float stdDev = (float)Math.Sqrt (variance);
863 
864  lowerBound = mean - stdDev * 6;
865  upperBound = mean + stdDev * 6;
866  }
