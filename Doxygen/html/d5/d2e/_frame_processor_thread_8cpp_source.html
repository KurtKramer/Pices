305 LogicalFramePtr FrameProcessorThread::GetNextFrameToProcess ()
306 {
307  LogicalFramePtr logicalFrame = framePool->GetNextFrameToProcess ();

309  {
310  osSleep (0.01f);

322  logicalFrame = NULL;
323  }
