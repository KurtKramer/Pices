399  int zoomedLargestDim = (int)((float)largestDim * zoomFactor + 0.5f);
400 
401  if (zoomedLargestDim > 4096)

403  zoomFactor = 4096.0f / (float)(zoomedLargestDim + 1);

406  int smallestDim = Math.Min (raster.Height, raster.Width);
407  int zoomedSmalestDim = (int)((float)smallestDim * zoomFactor + 0.5f);
408  if (zoomedSmalestDim < 3)
409  {
410  zoomFactor = 3.0f / (float)(zoomedSmalestDim + 1);
