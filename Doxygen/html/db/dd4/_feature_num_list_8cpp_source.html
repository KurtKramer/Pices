488 {

497  KKStr field = parser.GetNextToken (",\t");
498  if (field.Empty ())
499  continue;

501  if (dashPos < 0)

510  }

515  kkint32 endNum = field.SubStrPart (dashPos + 1).ToInt32 ();
516 
517  if ((startNum > endNum) || (startNum < 0) || (endNum > uint16_max))
518  {

524  results->push_back (z);

531  results = NULL;
532  }
