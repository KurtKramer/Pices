342 void FrameExtractorThread::SearchForCleanBreak ()
343 {
344  // A Clean Break is defined by having three rows in a row that do not
345  // have any pixels occupied.
346 
347  lastRowInFrame = lastRowUsed;
348 
349  int blankRowsInARow = 0;
350 
351  if ((pixelsPerRow[lastRowInFrame] == 0) &&
352  (pixelsPerRow[lastRowInFrame - 1] == 0))
353  {
354  // The Frame ends with a clean break so lets set next Frame to
355  // frameHeightMax indicating that there is nothing to carry over
356  // and back lastRowInFrame and lastRowUsed to first non blank
357  // Row.
358 
359  lastRowInFrame--;
360  while ((lastRowInFrame > 100) && (pixelsPerRow[lastRowInFrame] == 0))
361  lastRowInFrame--;
362 
363  nextFrameStartRow = frameHeightMax;
364  lastRowUsed = lastRowInFrame;
365  }
366 
367  else
368  {
369  // Since the last row has some pixels we need to search back for
370  // a clean break in the frame.
371 
372  while ((lastRowInFrame > 1000) && (blankRowsInARow < 3))
373  {
374  if (pixelsPerRow[lastRowInFrame] == 0)
375  {
376  blankRowsInARow++;
377  }
378  else
379  {
380  blankRowsInARow = 0;
381  nextFrameStartRow = lastRowInFrame;
382  }
383 
384  lastRowInFrame--;
385  }
386 
387  if ((lastRowInFrame <= 1000) && (blankRowsInARow < 3))
388  {
389  // Since we could not find a separation between images lets look
390  // for were the row with least pixels.
391  kkuint32 rowWithLeastPixels = lastRowInFrame;
392  kkuint32 leastNumOfPixels = 999999;
393 
394  for (kkuint32 x = lastRowInFrame; x < lastRowUsed; ++x)
395  {
396  if (pixelsPerRow[x] < leastNumOfPixels )
397  {
398  rowWithLeastPixels = x;
399  leastNumOfPixels = pixelsPerRow[x];
400  }
401  }
402 
403  lastRowInFrame = rowWithLeastPixels;
404  nextFrameStartRow = lastRowInFrame + 1;
405  }
406  else
407  {
408  nextFrameStartRow = lastRowInFrame + 1;
409 
410  // We now have a Good Starting Position for the next Frame.
411  // So now lets look for the ending position of the prev frame.
412  while ((lastRowInFrame > 100) && (pixelsPerRow[lastRowInFrame] == 0))
413  lastRowInFrame--;
414  }
415  }
416 } /* SearchForCleanBreak */

461

469  col++;
470

475  endCol = col;
476  col++;
477  }

480  endColFound = true;
481  }
482  }

485 
486  if (width > MaxArtifactLineWidth)
487  continue;
488 
489  // Now that we defined a Possible Column lets get rid of the stupid Line.
490 
491  uchar* rowContent = NULL;
492  for (row = 0; row <= lastRowUsed; row++)
493  {
494  rowContent = frame[row];
495 
496  if (((rowContent[startCol - 1] == 0) || (startCol <= 1)) &&
497  ((rowContent[endCol + 1] == 0) || (endCol >= (frameWidth - 2))))
